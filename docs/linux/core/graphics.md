### Go Example: –ü–æ–ª—É—á–µ–Ω–∏–µ —Ä–∞–∑—Ä–µ—à–µ–Ω–∏—è —ç–∫—Ä–∞–Ω–∞ —á–µ—Ä–µ–∑ X11

–≠—Ç–æ—Ç –ø—Ä–∏–º–µ—Ä –ø–æ–∫–∞–∑—ã–≤–∞–µ—Ç, –∫–∞–∫ Go-–ø—Ä–∏–ª–æ–∂–µ–Ω–∏–µ –º–æ–∂–µ—Ç –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤–æ–≤–∞—Ç—å —Å X-—Å–µ—Ä–≤–µ—Ä–æ–º (–∏—Å–ø–æ–ª—å–∑—É—è –±–∏–±–ª–∏–æ—Ç–µ–∫—É `xgb`) –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏–∏ –æ —Ç–µ–∫—É—â–µ–º —Ä–∞–∑—Ä–µ—à–µ–Ω–∏–∏ —ç–∫—Ä–∞–Ω–∞. –≠—Ç–æ –¥–µ–º–æ–Ω—Å—Ç—Ä–∏—Ä—É–µ—Ç –∫–ª–∏–µ–Ω—Ç-—Å–µ—Ä–≤–µ—Ä–Ω—É—é –ø—Ä–∏—Ä–æ–¥—É X11.

```go
// main.go
package main

import (
	"fmt"
	"log"

	"github.com/jezek/xgb"
	"github.com/jezek/xgb/xproto"
)

func main() {
	// –ü–æ–¥–∫–ª—é—á–∞–µ–º—Å—è –∫ X-—Å–µ—Ä–≤–µ—Ä—É. –ü–µ—Ä–µ–º–µ–Ω–Ω–∞—è DISPLAY –±—É–¥–µ—Ç –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∞ –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏.
	X, err := xgb.NewConn()
	if err != nil {
		log.Fatalf("Failed to connect to X server: %v", err)
	}
	defer X.Close()

	// –ü–æ–ª—É—á–∞–µ–º –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—é –æ –∫–æ—Ä–Ω–µ–≤–æ–º –æ–∫–Ω–µ (root window)
	setup := xproto.Setup(X)
	root := setup.DefaultScreen(X).Root

	// –ó–∞–ø—Ä–∞—à–∏–≤–∞–µ–º –≥–µ–æ–º–µ—Ç—Ä–∏—é –∫–æ—Ä–Ω–µ–≤–æ–≥–æ –æ–∫–Ω–∞
	geom, err := xproto.GetGeometry(X, xproto.Drawable(root)).Reply()
	if err != nil {
		log.Fatalf("Failed to get geometry: %v", err)
	}

	fmt.Printf("Screen resolution: %d x %d\n", geom.Width, geom.Height)
}
```
